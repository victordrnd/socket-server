106 {

110 
111  /* create socket */
112  sockfd = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
113  if (sockfd <= 0)
114  {
115  fprintf(stderr, "error: cannot create socket\n");
116  return -3;
117  }
118 
119  address.sin_family = AF_INET;
120  address.sin_addr.s_addr = inet_addr(configuration->bind_ip);
121  address.sin_port = htons(port);
122 
123  /* prevent the 60 secs timeout */
124  int reuse = 1;
125  setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, (const char *)&reuse, sizeof(reuse));
126  int flags = 1;
127  setsockopt(sockfd, SOL_SOCKET, TCP_NODELAY, (void *)&flags, sizeof(flags));
128  /* bind */
129  if (bind(sockfd, (struct sockaddr *)&address, sizeof(struct sockaddr_in)) < 0)
130  {
131  fprintf(stderr, "error: cannot bind socket to port %d\n", port);
132  return -4;
133  }
134 
135  return sockfd;
