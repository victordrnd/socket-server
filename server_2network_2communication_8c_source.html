140  address.sin_port = htons(port);
141 
142  /* prevent the 60 secs timeout */
143  int reuse = 1;
144  setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, (const char *)&reuse, sizeof(reuse));
145  int flags = 1;

147  /* bind */

152  }
