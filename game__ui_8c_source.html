56 void activate_countdown(GtkProgressBar *progress_bar, int seconds, int fps)
57 {
58  gtk_widget_show((GtkWidget *)progress_bar);
59  gtk_progress_bar_set_show_text(progress_bar, TRUE);
60  for (int i = seconds * fps; i >= 0; i--)
61  {
62  gdouble fraction = (i * (1.0 / seconds)) / fps;
63  if ((i / (fps / 10)) % 10 == 0)
64  {
65  char *res = (char *)malloc(3 * sizeof(char));

67  gtk_progress_bar_set_text(progress_bar, res);
68  //free(res);
69  }
70  gtk_progress_bar_set_fraction(progress_bar, fraction);
71  while (gtk_events_pending())
72  gtk_main_iteration();

74  }
