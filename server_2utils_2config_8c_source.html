28 void parse_game_configuration(GameConfiguration *game_configuration, config_setting_t *settings)
29 {
30  game_configuration->nb_room = config_setting_length(settings);
31  assert(game_configuration->nb_room > 0);
32  game_configuration->rooms = (Room *)malloc((int)game_configuration->nb_room * sizeof(Room));
33  Room *wp = game_configuration->rooms;
34  assert(sizeof(*wp) == sizeof(Room));
35  for (int i = 0; i < (int)game_configuration->nb_room; ++i)
36  {
37  config_setting_t *current_room_config = config_setting_get_elem(settings, i);
38  config_setting_lookup_string(current_room_config, "name", (const char **)&wp->name);
39  config_setting_lookup_int(current_room_config, "initial_amount", (int *)&wp->initial_amount);
40  config_setting_lookup_int(current_room_config, "nb_rounds", (int *)&wp->nb_rounds);
41 
42  assert(strlen(wp->name) > 1);
43  assert(wp->initial_amount > 10);
44  assert(wp->nb_rounds > 0);
45 
46  const config_setting_t *clients_room_config = config_setting_get_member(current_room_config, "clients");
47  const int nb_players = config_setting_length(clients_room_config);
48  wp->clients_id = (unsigned int *)malloc(nb_players * sizeof(int));
49  for (int j = 0; j < nb_players; j++)
50  {
51  const config_setting_t *client_name_config = config_setting_get_elem(clients_room_config, j);
52  config_setting_lookup_int(client_name_config, "client_id", (int *)&wp->clients_id[j]);
53  assert(wp->clients_id[j] > 0);
54  }
55  game_configuration->rooms[i] = *wp;
56  wp++;
57  }
58 }
