39 void parse_game_configuration(GameConfiguration *game_configuration, config_setting_t *settings)
40 {
41  game_configuration->nb_room = config_setting_length(settings);
42  assert(game_configuration->nb_room > 0);
43  game_configuration->rooms = (Room *)malloc((int)game_configuration->nb_room * sizeof(Room));
44  Room *wp = game_configuration->rooms;
45  assert(sizeof(*wp) == sizeof(Room));
46  for (int i = 0; i < (int)game_configuration->nb_room; ++i)
47  {
48  config_setting_t *current_room_config = config_setting_get_elem(settings, i);
49  config_setting_lookup_string(current_room_config, "name", (const char **)&wp->name);
50  config_setting_lookup_int(current_room_config, "initial_amount", (int *)&wp->initial_amount);
51  config_setting_lookup_int(current_room_config, "nb_rounds", (int *)&wp->nb_rounds);
52  config_setting_lookup_int(current_room_config, "waiting_time", (int *) &wp->waiting_time);
53  assert(strlen(wp->name) > 1);
54  assert(wp->initial_amount > 10);
55  assert(wp->nb_rounds > 0);
56 
57  const config_setting_t *clients_room_config = config_setting_get_member(current_room_config, "clients");
58  const int nb_players = config_setting_length(clients_room_config);
59  wp->clients_id = (unsigned int *)malloc(nb_players * sizeof(int));
60  for (int j = 0; j < nb_players; j++)
61  {
62  const config_setting_t *client_name_config = config_setting_get_elem(clients_room_config, j);
63  config_setting_lookup_int(client_name_config, "client_id", (int *)&wp->clients_id[j]);
64  assert(wp->clients_id[j] > 0);
65  }
66  game_configuration->rooms[i] = *wp;
67  wp++;
68  }
69 }

84  if (!config_read_file(&cfg, filename))
85  {

88  config_destroy(&cfg);
89  }
90  else
91  {
92  assert(sizeof(cfg) == sizeof(config_t));
93  config_lookup_string(&cfg, "bind_ip", &conf->bind_ip);
94  config_lookup_int(&cfg, "bind_port", (int *)&conf->bind_port);
95  config_lookup_int(&cfg, "max_simultaneous_connection", (int *)&conf->max_simultaneous_connection);
96 
97  setting = config_lookup(&cfg, "game_configuration");
98  GameConfiguration *game_configuration = (GameConfiguration *)malloc(sizeof(GameConfiguration));
