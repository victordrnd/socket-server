156 void on_action_received(Encapsulation *packet)
157 {
158  Game *game = (Game *)packet->data;
159  unsigned int opponent_id = get_opponent_id(packet->sender_id);
160  Session *opponent_session = get_client_session(opponent_id);
161  if (opponent_session != NULL)
162  {
163  Game *opponent_game = opponent_session->game;
164  calculate_balance(game, opponent_game);
165  Room *room = get_client_room(packet->sender_id);
166  write_line(room->name, game, packet->sender_id);
167  write_line(room->name, opponent_game, opponent_session->client_id);
168  send_round_end(packet->sender_id, game);
169  send_round_end(opponent_id, opponent_game);
170  remove_session(opponent_session);
171  if (game->current_round < get_max_round_count(room))
172  {
173  send_round_start(packet->sender_id, room);
174  send_round_start(opponent_id, room);
175  }
176  else
177  {
178  check_results(game, opponent_game, packet->sender_id, opponent_id);
179  }
180  }
181  else
182  {
183  Session current_session = {packet->sender_id, game};
184  add_session(&current_session); //save current player game awaiting opponent game.
185  }
186 }

194 void calculate_balance(Game *game, Game *opponent_game)
195 {
196  if (game->action == BETRAY)
197  {
198  if (opponent_game->action == BETRAY)
199  { //J1 Trahi et J2 Trahi
200  game->balance -= game->bet;
201  opponent_game->balance -= opponent_game->bet;
202  }
203  else
204  { //Collaborate
205  game->balance += game->bet;
206  opponent_game->balance -= opponent_game->bet;
207  }
208  }
209  else
210  {
211  if (opponent_game->action == BETRAY)
212  { //J1 Collabore et J2 Trahi
213  game->balance -= game->bet;
214  opponent_game->balance += opponent_game->bet;
215  }
216  }
217 }
