97 void on_action_received(Encapsulation *packet)
98 {
99  Game *game = (Game *)packet->data;
100  unsigned int opponent_id = get_opponent_id(packet->sender_id);
101  Session *opponent_session = check_opponent_played(opponent_id);
102  if (opponent_session != NULL)
103  {
104  Game *opponnent_game = opponent_session->game;
105  if (game->action == BETRAY)
106  {
107  if (opponnent_game->action == BETRAY)
108  { //J1 Trahi et J2 Trahi
109  game->balance -= game->bet;
110  opponnent_game->balance -= opponnent_game->bet;
111  }
112  else
113  { //Collaborate
114  game->balance += game->bet;
115  opponnent_game->balance -= opponnent_game->bet;
116  }
117  }
118  else
119  {
120  if (opponnent_game->action == BETRAY)
121  { //J1 Collabore et J2 Trahi
122  game->balance -= game->bet;
123  opponnent_game->balance += opponnent_game->bet;
124  }
125  }
126  Room *room = get_client_room(packet->sender_id);
127  write_line(room->name, game, packet->sender_id);
128  write_line(room->name, opponnent_game, opponent_session->client_id);
129  game->action = NOACTION;
130  opponnent_game->action = NOACTION;
131  send_round_end(packet->sender_id, game);
132  send_round_end(opponent_id, opponnent_game);
133  remove_session(opponent_session);
134  if (game->current_round < get_max_round_count(room))
135  {
136  send_round_start(packet->sender_id);
137  send_round_start(opponent_id);
138  }
139  else
140  {
141  check_results(game, opponnent_game, packet->sender_id, opponent_id);
142  }
143  }
144  else
145  {
146  Session current_session = {packet->sender_id, game};
147  add_session(&current_session);
148  }
149 }
