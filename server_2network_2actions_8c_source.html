149 void on_action_received(Encapsulation *packet)
150 {
151  Game *game = (Game *)packet->data;
152  unsigned int opponent_id = get_opponent_id(packet->sender_id);
153  Session *opponent_session = check_opponent_played(opponent_id);
154  if (opponent_session != NULL)
155  {
156  Game *opponent_game = opponent_session->game;
157  calculate_balance(game, opponent_game);
158  Room *room = get_client_room(packet->sender_id);
159  write_line(room->name, game, packet->sender_id);
160  write_line(room->name, opponent_game, opponent_session->client_id);
161  send_round_end(packet->sender_id, game);
162  send_round_end(opponent_id, opponent_game);
163  remove_session(opponent_session);
164  if (game->current_round < get_max_round_count(room))
165  {
166  send_round_start(packet->sender_id, room);
167  send_round_start(opponent_id, room);
168  }
169  else
170  {
171  check_results(game, opponent_game, packet->sender_id, opponent_id);
172  }
173  }
174  else
175  {
176  Session current_session = {packet->sender_id, game};
177  add_session(&current_session); //save current player game awaiting opponent game.
178  }
179 }

189 void calculate_balance(Game *game, Game *opponent_game){
190  if (game->action == BETRAY)
191  {
192  if (opponent_game->action == BETRAY)
193  { //J1 Trahi et J2 Trahi
194  game->balance -= game->bet;
195  opponent_game->balance -= opponent_game->bet;
196  }
197  else
198  { //Collaborate
199  game->balance += game->bet;
200  opponent_game->balance -= opponent_game->bet;
201  }
202  }
203  else
204  {
205  if (opponent_game->action == BETRAY)
206  { //J1 Collabore et J2 Trahi
207  game->balance -= game->bet;
208  opponent_game->balance += opponent_game->bet;
209  }
210  }
